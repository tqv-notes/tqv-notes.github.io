$$
y_r = \left|\langle a_r, z \rangle \right|^2, ~~ \text{for}~ r = 1, 2, ..., m
$$

$$
L(z) = \frac{1}{2m} \sum_{r=1}^{m} \left(y_r - \left|\langle a_r, z \rangle \right|^2\right)^2
$$

$$
Y = \sum_{r=1}^{m} y_r a_r a_r^*
$$
